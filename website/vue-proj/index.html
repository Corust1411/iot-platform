import { createRouter, createWebHistory } from 'vue-router'; import Login from '@/views/Login.vue'; import Dashboard from '@/views/Dashboard.vue'; const routes = [ { path: '/', component: Login }, { path: '/dashboard', component: Dashboard } ]; const router = createRouter({ history: createWebHistory(), routes }); router.beforeEach((to, from, next) => { const token = localStorage.getItem('token'); if (to.path !== '/' && !token) { next('/'); } else { next(); } }); export default router;
